133:      * 
134:      * @param boolean $max Set to true to retrieve maximum in/out rather than average.
135:      * 
136:      * @return string JSON encoded data
137:      */
138:     public function retrieveLog($max=false) {
139:         
140:         // arrays for parsed data
141:         $in = $out = $stamps = array();
142:         
143:         foreach (file("{$this->name}.log") as $line) {
144:             
145:             // ignore the summary line
146:             if (!isset($header)) {
147:                 $header = $line;
148:                 continue;
149:             }
150:             
151:             $parts = explode(' ', rtrim($line));
152:             if ($parts[1] == 0 && $parts[2] == 0 && $parts[3] == 0 && $parts[4] == 0) continue;
153:             
154:             array_push($stamps, $parts[0]);
155:             
156:             if ($max) {
157:                 $in[] = round($parts[3]/1024);
158:                 $out[] = round($parts[4]/1024);
159:             } else {
160:                 $in[] = round($parts[1]/1024);
161:                 $out[] = round($parts[2]/1024);
162:             }
163:             
164:         }
165: 
166:         // determine earliest and latest timestamps
167:         $latest = array_shift($stamps);
168:         $earliest = array_pop($stamps);
169:         
170:         // calculate average interval
171:         $interval = round(($latest - $earliest) / count($stamps)+2, 0);
172:         
173:         // encode and return response
174:         return json_encode(array(
175:             'earliest'  => $earliest,
176:             'latest'    => $latest,
177:             'interval'  => $interval,
178:             'intervalMin'   => round($interval/60, 0),
179:             'inData'    => array_reverse($in),
180:             'outData'   => array_reverse($out)
181:         ));
182:         
183:     }
184: 
185: }
